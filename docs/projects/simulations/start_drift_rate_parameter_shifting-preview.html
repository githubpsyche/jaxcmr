def save_figure(figure_dir, figure_str, suffix=None):
    plt.tight_layout()
    if not figure_str:
        plt.show()
        return
    os.makedirs(figure_dir, exist_ok=True)
    suffix_str = f"_{suffix}" if suffix else ""
    figure_path = os.path.join(figure_dir, f"{figure_str}{suffix_str}.png")
    plt.savefig(figure_path, bbox_inches="tight", dpi=600)
    plt.show()

# generate figures comparing model and data
for analysis_cfg in comparison_analyses:
    analysis_fn = analysis_cfg["target"]
    analysis_suffix = analysis_cfg["figure_suffix"]
    figure_path = (
        os.path.join(
            figure_dir,
            f"{figure_str}_{analysis_suffix}_shift_{varied_parameter}.png",
        )
        if figure_str
        else None
    )
    if figure_path:
        print(f"![]({figure_path})")

    if figure_path and os.path.exists(figure_path) and not redo_figures:
        display(Image(filename=figure_path))
        continue

    # Create figure (slightly narrower)
    fig, axis = plt.subplots(figsize=(8, 5))
    
    # Use distinct colors for 3 lines
    color_cycle = ["#AAAAAA", "#555555", "#000000"]
    
    # Use interpretable labels
    line_labels = ["None [0]", "Medium [.45]", "High [.9]"]

    trial_mask = generate_trial_mask(data, trial_query)
    sim_trial_mask = generate_trial_mask(sim[0], trial_query)

    base_kwargs = {
        "datasets": sim,
        "trial_masks": [np.array(sim_trial_mask)] * len(considered_values),
        "color_cycle": color_cycle,
        "labels": line_labels,
        "contrast_name": varied_parameter,
        "axis": axis,
    }
    base_kwargs |= analysis_cfg["kwargs"]

    signature = inspect.signature(analysis_fn)
    filtered_kwargs = {
        name: value
        for name, value in base_kwargs.items()
        if name in signature.parameters
    }

    axis = analysis_fn(**filtered_kwargs)
    
    # Make lines progressively thicker
    lines = axis.get_lines()
    for i, line in enumerate(lines):
        line.set_linewidth(1.5 + i * 0.5)

    # Format the plot (font sizes, legend location, etc.)
    axis.tick_params(labelsize=14)
    axis.set_xlabel(axis.get_xlabel(), fontsize=16)
    axis.set_ylabel(axis.get_ylabel(), fontsize=16)
    axis.legend(loc="upper left", bbox_to_anchor=(1.02, 1), title="Start Reinstatement")
    
    # Add vertical line separating "Film" and "Interference" regions
    boundary = 13
    axis.axvline(x=boundary, color="red", linewidth=2, linestyle="--", alpha=0.7)
    
    # Add region labels at top of plot
    ylim = axis.get_ylim()
    xlim = axis.get_xlim()
    label_y = ylim[1] + 0.02 * (ylim[1] - ylim[0])
    
    # "Film" label centered in left region
    film_center = (xlim[0] + boundary) / 2
    axis.text(film_center, label_y, "Film", ha="center", va="bottom", 
              fontsize=14, fontweight="bold", color="black")
    
    # "Interference" label centered in right region  
    interference_center = (boundary + xlim[1]) / 2
    axis.text(interference_center, label_y, "Tetris", ha="center", va="bottom",
              fontsize=14, fontweight="bold", color="red")
    
    # Add annotation to the right of the plot, below the legend
    fig.text(
        0.88, 0.45,  # Position in figure coordinates (right side, middle-ish)
        "Beginning recall at\nstart-of-list context\nhelps focus on target items\nand suppress interference\nfrom later experiences\nin voluntary free recall.",
        fontsize=11,
        ha="center",
        va="top",
        style="italic",
    )

    if analysis_cfg["ylim"] is not None:
        axis.set_ylim(analysis_cfg["ylim"])

    save_figure(
        figure_dir,
        figure_str,
        suffix=f"{analysis_suffix}_shift_{varied_parameter}",
    )
